% Creation de boite TODO, qui apparaissent dans les signets

    \stepcounter{TODO}
    \hypertarget{TODO\theTODO}{}%
    \begin{tcolorbox}[%
            title={TODO \theTODO},%
            colback=red!30!white,%
            colframe=red!75!black,%
            halign=flush center%
        ]
        #1%
    \end{tcolorbox}%
    \pdfbookmark[0]{TODO \theTODO}{TODO\theTODO}%
}
